    # 图片转 base64 上传

    # 本地图片转 base64

    1. 新建文件封装异步转 base64 方法
    export function uploadImgToBase64(file) {
  return new Promise((resolve, reject) => {
    const reader = new FileReader();
    reader.readAsDataURL(file);
    reader.onload = function () {
      // 图片转base64完成后返回reader对象
      resolve(reader);
    };
    reader.onerror = reject;
  });
}
    1. 添加商品的时候选择本地文件，选中用对象保存整个 file 对象
    fileId.onchange = function (e) {
  console.log(e);
  let file = e.target.files[0]; //file文件
  uploadImgToBase64(file);
};
    1. 最后提交整个商品表单之前进行编码处理

    # 在线文件转 base64

    1. utils 文件添加在线文件转 base64 方法，利用 canvas
    export function URLImgToBase64(img) {
  return new Promise((resolve, reject) => {
    function getBase64Image(img) {
      const canvas = document.createElement("canvas");
      canvas.width = img.width;
      canvas.height = img.height;
      const ctx = canvas.getContext("2d");
      ctx.drawImage(img, 0, 0, canvas.width, canvas.height);
      var dataURL = canvas.toDataURL();
      return dataURL;
    }

    const image = new Image();
    image.crossOrigin = "*"; // 允许跨域图片
    image.src = img + "?v=" + Math.random(); // 清除图片缓存
    image.onload = function () {
      resolve(getBase64Image(image));
    };
    image.onerror = reject;
  });
}
    1. 编辑商品，先拉取原来的商品信息展示到页面上
    2. 提交表单之前，区分在线图片还是本地图片进行转码

    # 图片压缩上传

    实现的原理就是利用 canvas 的 toDataURL 来进行比例压缩，压缩后是一个 base64 格式的数据

    # 1. 获取 file 文件

    fileId.onchange = function (e) {
  console.log(e);
  let file = e.target.files[0]; //file文件
  fileToBase64(file, 0.7);
};

    # 2. 获取到图片文件类型

    let type = file.type; // "image/png"

    # 3. file 文件转成 base64 格式

    const fileToBase64 = (file, quality) => {
  //图片对象      图片质量
  let fileReader = new FileReader(); // // 创建FileReader对象
  let type = file.type; // "image/png"
  // 对文件进行读取
  fileReader.readAsDataURL(file); // 传入参数：file文件
  fileReader.onload = function (e) {
    console.log("原始二进制字符串：", this.result.toString());
    compress(fileReader.result, quality, type); // 输出参数：文件的base64数据
  };
};

    注意点：触发 FileReader 的 onload 事件，转换 base64 完毕返回 result 属性包含一个 data:URL 格式的字符串（base64 编码）以表示所读取文件的内容

    # 4. 通过 canvas 对 base64 数据进行压缩

    1. 传入参数：base64 文件、文件类型
    2. base64 压缩：Canvas.toDataURL(mimeType, quality)

      传入参数：类型（默认 image/png），压缩度（0-1，默认 0.92）

      返回值：压缩后的 base64 格式的数据

    Canvas.getContext("2d").drawImage(img, 0, 0, cvs.width, cvs.height);

    而且需要设置 crossOrigin，因为 canvas 画布内的图片可能会存在跨域问题，需要处理，否则可能会造成渲染失败

    预览的话，也可以做图片大小与 canvas 大小的适配

    if (img.width > MAX_WIDTH) {

    # 5. 将 base64 数据转位 blob 二进制

    # 解法一：converrBase64UrlToBlob(base64, mimeType)

    1. window.atob进行解码 base64 格式的数据
    let bytes = window.atob(base64.split(",")[1]); //atob方法用于解码base64
    1. 通过ArrayBuffer创建定制长度的 buffer（二进制）文件
    // 创建一个长度为 bytes.length 的 buffer(一个二进制文件), 它会分配一个 16 字节（byte）的连续内存空间，并用 0 进行预填充。
let ab = new ArrayBuffer(bytes.length);

    ArrayBuffer 对象是用来表示通用的、固定长度的原始二进制数据缓冲区

    它是一个字节数组，通常在其他语言中称为“byte array”

    1. 通过Unit8Array更改初始的二进制文件
    // Uint8Array —— 将 ArrayBuffer 中的每个字节视为 0 到 255 之间的单个数字（每个字节是 8 位）。这称为 “8 位无符号整数”。
let ia = new Uint8Array(ab);
for (let i = 0; i < bytes.length; i++) {
  // 更改里面的初始化内容
  ia[i] = bytes.charCodeAt(i);
}
    1. 将处理后的数据塞入一个新的 Blob 对象
    // 创建blob格式数据，并传入二进制文件和文件原本类型
let _blob = new Blob([ia], { type: mimeType });
toImg(_blob);
return _blob;

    # 解法二：canvas.toBlob(callback, mimeType, quality)

    canvas.toBlob(
  (blob) => {
    if (blob) {
      // 返回的blob对象
      // 使用 Blob 对象（例如上传或保存）
      const formData = new FormData();
      formData.append("image", blob, "image.png");
      fetch("/upload", { method: "POST", body: formData });
    } else {
      console.error("生成 Blob 失败");
    }
  },
  "image/jpeg",
  0.8
); // 格式为 JPEG，质量 80%

    image/png（默认，无损压缩） image/jpeg（有损压缩，可指定质量） image/webp（现代浏览器支持，质量/压缩比更优）

    toBlob 是异步操作，不会阻塞主线程。处理大画布或者复杂内容时，建议始终等待回调完成。

    如果画布被跨域图像污染（如加载了不同域的图片），调用 toBlob()会失败，需确保图片资源允许跨域访问（CORS 配置img.crossOrigin = "anonymous";）

    # 总结

    图片压缩流程：

    获取 file->创建 FileReader->FileReader 的 readAsDataURL 转换成 base64->创建 canvas->canvas.toDataURL(mimeType, quality)压缩 base64->canvas.toBlob 转成 blob 二进制（或->new File(blob)转成 file 进行分片）

    # 补充 - 已上传的图片处理

    解法一：用 canvas.drawImage 绘制显示图片，注意要进行跨域处理 crossOrigin:"anonymous"

    解法二：URL.createObjectURL - 创建 Blob 或者 File 的临时 url，可以用来本地预览图片

    删除已上传的图片：

    解法一：上传所有的图片数组（不推荐，好多啊～而且已上传的为什么还要再上传一遍，傻吗）

    解法二：上传删除操作后的新数组（只存放 file.name）

    解法三：给删除的图片加标识，表示该图片要被删除
