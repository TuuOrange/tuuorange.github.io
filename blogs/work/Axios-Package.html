    # 为什么要进行 axios 封装

    对 axios 进行封装可以统一处理我们的 http 请求，通过请求响应拦截，统一处理我们的请求、加载和错误处理

    # 封装 axios 实现的功能

    • 请求拦截前的 loading、请求头和参数配置
    • 发起请求之后取消重复请求
    • 响应之后的错误处理和数据包装

    # 请求拦截

    请求拦截主要处理 loading 和请求头，以及配置用户标识（token）

    // 处理请求头
const handleRequestHeader = (config) => {
  config["xxxx"] = "xxx";
  return config;
};
// 配置用户标识
const handleAuth = (config) => {
  config.header["token"] = localStorage.getItem("token") || token || "";
  return config;
};
// 请求拦截器
axios.interceptors.request.use((config) => {
  // 处理config
  config = handleChangeRequestHeader(config);
  config = handleConfigureAuth(config);
  /* loading */
  return config;
});

    # 响应拦截

    响应拦截主要处理报错，有三种报错类型

    • 网络请求错误
    • 用户授权错误
    • 成功之后返回的提示的普通的错误
    /* src/api/tool.ts */
// 网络请求错误
const handleNetworkError = (errStatus) => {
  let errMessage = "未知错误";
  if (errStatus) {
    switch (errStatus) {
      case 400:
        errMessage = "错误的请求";
        break;
      case 401:
        errMessage = "未授权，请重新登录";
        break;
      case 403:
        errMessage = "拒绝访问";
        break;
      case 404:
        errMessage = "请求错误,未找到该资源";
        break;
      case 405:
        errMessage = "请求方法未允许";
        break;
      case 408:
        errMessage = "请求超时";
        break;
      case 500:
        errMessage = "服务器端出错";
        break;
      case 501:
        errMessage = "网络未实现";
        break;
      case 502:
        errMessage = "网络错误";
        break;
      case 503:
        errMessage = "服务不可用";
        break;
      case 504:
        errMessage = "网络超时";
        break;
      case 505:
        errMessage = "http版本不支持该请求";
        break;
      default:
        errMessage = `其他连接错误 --${errStatus}`;
    }
  } else {
    errMessage = `无法连接到服务器！`;
  }

  message.error(errMessage);
};
// 用户授权错误
const handleAuthError = (errStatus) => {
  const authErrMap: any = {
    10031: "登录失效，需要重新登录", // token 失效
    10032: "您太久没登录，请重新登录~", // token 过期
    10033: "账户未绑定角色，请联系管理员绑定角色",
    10034: "该用户未注册，请联系管理员注册用户",
    10035: "code 无法获取对应第三方平台用户",
    10036: "该账户未关联员工，请联系管理员做关联",
    10037: "账号已无效",
    10038: "账号未找到",
  };

  if (authErrMap.hasOwnProperty(errStatus)) {
    message.error(authErrMap[errStatus]);
    // 授权错误，登出账户
    logout();
    return false;
  }

  return true;
};
// 200之后服务器返回的提示的普通的错误
const handleGeneralError = (errStatus, errmsg) => {
  if (err.errStatus !== "0") {
    meessage.error(err.errmsg);
    return false;
  }

  return true;
};

    # 响应拦截器中处理错误

    // 响应拦截器（错误返回Promise错误回调，成功返回数据）
axios.interceptors.response.use(
  (response) => {
    // 网络请求成功之后返回的错误
    if (response.status !== 200) return Promise.reject(response.data);

    handleAuthError(response.data.errStatus); // 用户授权错误
    handleGeneralError(response.data.errStatus, response.data.errmsg); // 普通错误（比如说参数不正确等）

    return response;
  },
  (err) => {
    // 网络请求错误
    handleNetworkError(err.response.status);
    Promise.reject(err.response);
  }
);

    # 处理 get 请求

    若我们想一直以const [err, result] = await api.getUserInfo(id)的方式调用，我们需要保证稳定的返回数据[err, result]，就代表不管是请求成功还是失败，我们都需要通过resole([err, result])调用返回数据

    type Fn = (data: FcResponse<any>) => unknown

interface IAnyObj {
  [index: string]: unknown
}

interface FcResponse<T> {
  errStatus: string
  errmsg: string
  data: T
}

const get = <T,>(url: string, params: IAnyObj = {}, clearFn?: Fn): Promise<[any, FcResponse<T> | undefined] => {
  new Promise((resolve) => {
    axios
      .get(url, { params })
      .then((result) => {
        let res: FcResponse<T>
        if (clearFn !== undefined) { // 对返回数据进行派生
          res = clearFn(result.data) as unknown as FcResponse<T>
        } else { // 直接返回
          res = result.data as FcResponse<T>
        }
        resolve([null, res as FcResponse<T>])
      })
      .catch((err) => {
        resolve([err, undefined])
      })
  })
}

    # 取消重复请求

    取消重复请求有两种方法，一个是浏览器提供的 AbortController，一个是 axios 自带的 cancelToken 对象

    # AbortController

    const controller = new AbortController();

axios
  .get("url", {
    signal: controller.signal,
  })
  .then((response) => {})
  .catch((error) => {
    if (error.name === "AbortError") {
      // 请求取消
      resolve(["cancel", undefined]);
    }
  });

// 取消请求
controller.abort();

    # cancelToken

    cancelToken 写法与 abort 类似

    const CancelToken = axios.CancelToken;
const source = CancelToken.source();

axios
  .get("url", {
    cancelToken: source.token,
  })
  .then((response) => {})
  .catch((error) => {
    if (axios.isCancel(error)) {
      resolve(["cancel", undefined]);
    }
  });
// 取消请求（message 参数是可选的）
source.cancel("Operation canceled by the user.");
